% ---------------------------------------------------------
\begin{frame}{Systemic \& Operational Challenges}
    \begin{alertblock}{Architectural Constraints}
        Key operational hurdles encountered during the design, development, and deployment phases of the churn platform.
    \end{alertblock}
    \begin{itemize}
        \item \textbf{Class Imbalance Skew:} The minor distribution of the positive churn class ($\approx 20.3\%$) inherently biases traditional objective functions toward the dominant negative class.
        \item \textbf{Target Feature Leakage:} Highly predictive indicators—such as customer complaint flags—occur downstream of the initial churn signal, risking data contamination if not strictly isolated.
        \item \textbf{Interpretability vs. Capacity Trade-off:} High-capacity tree ensembles (XGBoost) yield superior non-linear boundaries but sacrifice clear, parametric explainability.
    \end{itemize}
\end{frame}

% ---------------------------------------------------------
% NEW SLIDE: MITIGATION STRATEGIES
% ---------------------------------------------------------
\begin{frame}{Implemented Engineering Mitigations}
    \begin{exampleblock}{Algorithmic \& Pipeline Safeguards}
        How the platform resolved systemic challenges to enforce production-grade reliability.
    \end{exampleblock}
    \begin{itemize}
        \item \textbf{Cost-Sensitive Objective Weighting:} Injected an optimized \texttt{scale\_pos\_weight = 1.19} parameter into the booster to force the loss function to penalize minority-class misclassifications.
        \item \textbf{Proactive Data Isolation:} Formally stripped downstream target-leaking indicators from the registry configuration prior to modeling to guarantee an honest out-of-sample test setup.
        \item \textbf{Post-Hoc SHAP Layering:} Integrated localized tree-path dependent Shapley explanations to provide fully auditable, feature-level attribution without sacrificing model complexity.
    \end{itemize}
\end{frame}
