\begin{frame}{Advanced Modeling Strategy \& Hyperparameter Optimization}
\scriptsize
\begin{block}{Bayesian Search Framework}
Transitioned from generic grids to an intelligent \textbf{Tree-structured Parzen Estimator (TPE)} optimization loop tracking \textbf{Average Precision (PR-AUC)} over 500 parallel trials.
\end{block}

\vspace{0.1cm}

\begin{columns}[T]
    \begin{column}{0.48\textwidth}
        \textbf{Pipeline Search Dimensions}
        \begin{itemize}
\scriptsize
            \setlength{\itemsep}{2pt}
            \item \textbf{Preprocessing Adaptive Space:} Dynamic mapping between \texttt{Standard}, \texttt{MinMax}, and \texttt{Robust} scalers per trial.
            \item \textbf{Random Forest Space:} Tuning depth bounds ($8 \le d \le 10$) and split restrictions ($18 \le s \le 28$).
            \item \textbf{XGBoost Search Space:} Tuning scaling boundaries ($1.8 \le \text{pos\_weight} \le 2.2$), custom $\gamma$, and regularization parameters.
        \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
        \textbf{Computational Efficiency}
        \begin{itemize}
\scriptsize
            \setlength{\itemsep}{2pt}
            \item \textbf{Early Stopping Pruner:} Used a \texttt{MedianPruner} to instantly kill sub-optimal asset distributions at step zero.
            \item \textbf{Cross-Validation Protocol:} Strict 5-Fold \texttt{StratifiedKFold} setup wrapped within the objective scope to eliminate target leakage.
        \end{itemize}
    \end{column}
\end{columns}

\end{frame}

\textbf{Identified Action Items for Model Deployment Iterations:}
\vspace{0.1cm}
\begin{itemize}
    \small
    \setlength{\itemsep}{4pt}
    \item \textbf{\texttt{move\_left} / \texttt{move\_right}}: Detects parameter boundary pressures (e.g., if max depth hits its range limits), automatically calculating direction shifts for the next search loop.
    \item \textbf{\texttt{narrow}}: Shrinks bounds around optimal hotspots when hyperparameter values cluster closely together (concentration factor $< 0.40$), conserving compute.
    \item \textbf{\texttt{keep}}: Locks stable search limits for hyperparameter ranges that have safely converged, allowing deployment code to freeze with confidence.
\end{itemize}
