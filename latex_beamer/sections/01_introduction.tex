\begin{frame}{Business Problem}
\begin{itemize}

    \item Losing customers reduces deposits, transaction volume, and product cross-sell opportunities
    \item Early detection enables targeted retention strategies (offers, engagement, support)
    \item Objective: Predict probability of customer churn (\textit{Exited})
\end{itemize}
\end{frame}
